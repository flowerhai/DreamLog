**焦点陷阱实现**:
```javascript
// 记录上一个聚焦的元素
let lastFocusedElement = null;

// 可聚焦元素选择器
const FOCUSABLE_SELECTORS = 'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])';

// 获取模态框内所有可聚焦元素
function getFocusableElements(modal) {
    return modal.querySelectorAll(FOCUSABLE_SELECTORS);
}

// 焦点陷阱：在模态框内循环聚焦
function trapFocus(e, modal) {
    const focusableElements = getFocusableElements(modal);
    const firstElement = focusableElements[0];
    const lastElement = focusableElements[focusableElements.length - 1];
    
    if (e.key === 'Tab') {
        if (e.shiftKey) {
            if (document.activeElement === firstElement) {
                e.preventDefault();
                lastElement.focus();
            }
        } else {
            if (document.activeElement === lastElement) {
                e.preventDefault();
                firstElement.focus();
            }
        }
    }
}

// 打开模态框时
function openRecordModal() {
    lastFocusedElement = document.activeElement;
    recordModal.classList.add('active');
    
    // 聚焦到第一个可聚焦元素
    const focusableElements = getFocusableElements(recordModal);
    if (focusableElements.length > 0) {
        focusableElements[0].focus();
    }
    
    // 添加焦点陷阱
    recordModal.addEventListener('keydown', (e) => trapFocus(e, recordModal));
}

// 关闭模态框时
function closeRecordModal() {
    recordModal.classList.remove('active');
    recordModal.removeEventListener('keydown', trapFocus);
    
    // 恢复聚焦到之前的元素
    if (lastFocusedElement) {
        lastFocusedElement.focus();
        lastFocusedElement = null;
    }
}
```

## 🔧 技术要点

### 焦点陷阱 (Focus Trap)

**为什么需要**: 模态框打开时，防止 Tab 键聚焦到模态框外的元素，提升屏幕阅读器用户体验。

**实现方式**:
1. 监听 Tab 键事件
2. 检测当前聚焦元素是否为第一个/最后一个
3. 如果是，阻止默认行为并聚焦到另一端

### 焦点管理 (Focus Management)

**为什么需要**: 模态框关闭后，恢复聚焦到之前操作的元素，保持用户的上下文感知。

**实现方式**:
1. 打开模态框时记录 `document.activeElement`
2. 关闭模态框时恢复聚焦

### 键盘导航 (Keyboard Navigation)

**为什么需要**: 部分用户无法使用鼠标，需要支持纯键盘操作。

**实现方式**:
1. 为可点击元素添加 `tabindex="0"`
2. 监听 `keydown` 事件 (Enter/Space)
3. 阻止默认行为并触发点击
