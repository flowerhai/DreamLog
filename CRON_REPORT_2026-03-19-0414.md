```swift
// 挑战实时活动启动
try await liveActivityService.startChallengeActivity(challenge: challenge)

// 挑战进度更新

    challengeId: challengeId.uuidString, 
    challenge: challenge
)

// 挑战完成

    challengeId: challengeId.uuidString, 
    reason: .completed
)
```
