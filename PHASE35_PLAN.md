## 🔧 技术实现细节

### Core ML 模型训练

```swift
// 使用 Create ML 训练模型
import CreateML

// 1. 准备训练数据
let trainingData = try MLDataTable(contentsOf: trainingURL)

// 2. 训练情绪预测模型
let emotionModel = try MLRegressor(
    trainingData: trainingData,
    targetColumn: "emotion_score",
    modelParameters: [
        .treeEnsemble(
            parameters: .init(
                treeCount: 50,
                maxDepth: 10
            )
        )
    ]
)

// 3. 导出模型
try emotionModel.write(to: modelURL)
```

### 性能优化技巧

```swift
// 1. 延迟加载
@State private var dreams: [Dream] = []

var body: some View {
    List {
        ForEach(dreams) { dream in
            DreamRow(dream: dream)

    }
    .onAppear {
```
