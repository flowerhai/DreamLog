## 🔧 技术细节

### RootView 架构

```swift
struct RootView: View {
    @State private var showOnboarding = false
    
    var body: some View {
        Group {
            if showOnboarding {
                DreamOnboardingView()
            } else {
                ContentView()
            }
        }
        .onAppear {
            // 检查是否首次启动
            let isFirstLaunch = UserDefaults.standard.object(forKey: "hasCompletedOnboarding") == nil
            if isFirstLaunch {
                showOnboarding = true
            }
        }
    }
}
```
