## 📊 数据模型

### 核心模型

```swift
@Model
class MeditationSession {
    var id: UUID
    var type: MeditationType
    var duration: TimeInterval
    var completed: Bool
    var moodBefore: String?
    var moodAfter: String?
    var dreamId: UUID?
    var createdAt: Date
    var completedAt: Date?
}

@Model
class MeditationPreference {
    var id: UUID
    var favoriteTypes: [MeditationType]
    var preferredDuration: TimeInterval
    var preferredVoice: String
    var reminderTime: Date?
    var backgroundSound: String?
}

enum MeditationType: String, Codable, CaseIterable {
    // 引导冥想
    case guidedDream = "guided_dream"
    case sleepStory = "sleep_story"
    case lucidDreamInduction = "lucid_induction"
    
    // 呼吸练习
    case breathing478 = "breathing_478"
    case boxBreathing = "box_breathing"
    case wildBreathing = "wild_breathing"
    case morningWake = "morning_wake"
    
    // 放松扫描
    case bodyScan = "body_scan"
    case dreamRecall = "dream_recall"
    case stressRelief = "stress_relief"
    
    // 正念冥想
    case dreamAwareness = "dream_awareness"
    case realityCheck = "reality_check"
    case emotionalAwareness = "emotional_awareness"
    case gratitude = "gratitude"
    
    // 音乐疗法
    case musicTherapy = "music_therapy"
    case whiteNoise = "white_noise"
    case binauralBeats = "binaural_beats"
    case customMix = "custom_mix"
}
```
